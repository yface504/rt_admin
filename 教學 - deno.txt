# 從 Deno 中使用 Deno.run 方法來調用外部 Python 腳本
const process = Deno.run({
    cmd: ["python", "your_script.py"],
    stdout: "piped",
    stderr: "piped"
});

const output = await process.output(); // 捕獲標準輸出
const error = await process.stderrOutput(); // 捕獲標準錯誤

process.close(); // 關閉進程
